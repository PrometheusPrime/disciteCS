Inheritance

By default all classes are final and do not allow inheriting from them. Use the open keyword to make a class inheritable. All classes inherit from the top level class Any.

open class Parent(firstName: String,familyName : String){
  var name : String
  init {
      name = "$firstName $familyName"
  }
  fun info(): String {
        return "Name : $name"

    }
  
}
class Child(firstName: String, familyName: String, parent: Parent) : Parent(firstName, familyName){

}

//in main
val john = Parent("John", "Mwanza")
val moses = Child("Moses", "Mwanza",john)
println(moses.info())

Method Overriding

Methods and properties must

open class Parent(firstName: String, familyName: String) {
    var name: String

    init {
        name = "$firstName $familyName"
    }

    open fun info(): String {
        return "Name : $name"

    }

}

class Child(firstName: String, familyName: String, val parent: Parent) : Parent(firstName, familyName) {
    override fun info(): String {
        return "Name : $name\nParent : ${parent.name}"

    }
}

//in main
val john = Parent("John", "Mwanza")
val moses = Child("Moses", "Mwanza",john)

println(john.info())
println()
println(moses.info())
