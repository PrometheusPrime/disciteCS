Abstract Function

All abstract functions must be implemented in the derived classes.

abstract class Bird() {
    abstract fun canFly(): Boolean

}

class Palaeognathae() : Bird() {
    override fun canFly(): Boolean {
        return false
    }
}

class Neognathae() : Bird() {
    override fun canFly(): Boolean {
        return true
    }
}
