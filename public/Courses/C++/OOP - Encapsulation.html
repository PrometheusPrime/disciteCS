Encapsulation

It it not recommended to directly access and modify fields in a class. Make fields private and use setter and getter Methods to modify and access them.

class Planet {
private:
    string name;
    double surfaceArea;
    int numberOfSatellites;
    double distanceFromTheSun;
public:
// setters used to set values to the fields
    void setName(string pName)
    {
        name = pName;
    }
    void setSurfaceArea(double pSurfaceArea)
    {
        surfaceArea = pSurfaceArea;
    }
    void setNumberOfSatellites(int pNumberOfSatellites)
    {
        numberOfSatellites = pNumberOfSatellites;
    }
    void setDistanceFromTheSun(double pDistanceFromTheSun)
    {
        distanceFromTheSun = pDistanceFromTheSun;
    }
// getters used to get values of the fields
    string getName()
    {
        return name;
    }
    double getSurfaceArea()
    {
        return surfaceArea;
    }
    int getNumberOfSatellites()
    {
        return numberOfSatellites;
    }
    double getDistanceFromTheSun()
    {
        return distanceFromTheSun;
    }
// Constructors
    Planet()
    {
    }

    Planet(string name, double surfaceArea, int numberOfSatellites, double distanceFromTheSun)
    {

        setName(name);
        setSurfaceArea(surfaceArea);
        setNumberOfSatellites(numberOfSatellites);
        setDistanceFromTheSun(distanceFromTheSun);
    }
};

int main(int argc, char const *argv[])
{
    Planet uranus("Uranus", 8115600000, 27, 2870972200);
    cout << uranus.getName();

  return 0;
}
