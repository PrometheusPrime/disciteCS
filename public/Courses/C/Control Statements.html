Nested if statements

int age = 22;
if(age > 18){
  if(age <= 25){
    printf("You're between 18 and 25 years of age\n");
  }
}
int age = 22;
if (age > 18 && age <= 25){
    printf("You're between 18 and 25 years of age\n");
}
