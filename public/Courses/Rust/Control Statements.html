Conditional Statements

Control Statements are used to run code under certain conditions

if statement

if(expression){
    //TODO
}
if(4 < 5){
    println!("4 is less than 5\n");
}

Nested if statements

let age = 22;
if(age > 18){
  if(age <= 25){
    println!("You're between 18 and 25 years of age\n");
  }
}
let age = 22;
if (age > 18 && age <= 25){
    println!("You're between 18 and 25 years of age\n");
}
